            .df-summary-grid {
              grid-template-columns: 1fr;
            }
          }
        `}
      </style>

      <div className="df-chat-header">
        <div className="df-header-left">
          <div className="df-header-avatar">D</div>

          <div className="df-header-title">
            <strong>Assistente Delfinance</strong>
            <p>Converse comigo para montar seu caminho de integração.</p>
          </div>
        </div>

        <div className="df-status">
          <span className="df-status-dot" />
          Online
        </div>
      </div>

      <div ref={chatBodyRef} className="df-chat-body">
        <Bubble>
          Olá! Vou te ajudar a encontrar o melhor caminho para integrar com a Delfinance.
        </Bubble>

        {questions.map((question, index) => {
          if (index > activeQuestionIndex) return null;

          const answer = answers[question.key];

          return (
            <div key={question.key}>
              <Bubble>{question.question}</Bubble>

              <Options question={question} />

              {answer && (
                <Bubble type="user">
                  {question.userText(answer)}
                </Bubble>
              )}
            </div>
          );
        })}

        {typing && <TypingBubble />}

        {activeQuestionIndex >= questions.length && isComplete && (
          <>
            <Bubble>
              Pronto. Com base nas suas respostas, montei um plano recomendado para sua integração.
            </Bubble>

            <PlanCard />
          </>
        )}

        <div ref={bottomRef} />
      </div>
    </div>
  );
};
